## For loop
```swift
for var i = 0; i < 10; i+=1 {
	print(i) 
} 
// error
// C-style for statement has been removed in Swift 3
```
```swift
for i in 0..<10 {
    print(i)
}

for i in 1...5 {
    print("\(i) Hi")
}

for _ in 1...5 {
    print("Hi")
}
// I can be omitted as _
```

## While loop
```swift
var count = 0;
while count < 100 {
    count += 1
}
print(count)
// 100
```

## Repeat-while
```swift
var i = 10
repeat {
    i -= 1
    print(i)
} while (i > 0)
```

## Break
```swift
for i in 1..<10 {
    if i > 5 {
        break
    }
    print(i)
}
```

## Comtinue
```swift
for i in 1...10 {
    if i % 2 == 0 {
        continue
    }
    print(i)
}
```
