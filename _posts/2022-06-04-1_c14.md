```c
#include <stdio.h>

int main(void)
{
    int bus[5];
    int i, n;
    n = sizeof(bus) / sizeof(int);

    puts("-----------------------");
    puts(" ** Int Array i/o ** ");
    puts("-----------------------");

    for(i = 0; i < n; i++)
    {
        printf(" index [%d] bus No. : ", i);
        scanf("%d", &bus[i]);
    }

    printf(" \n * There are [%d] bus * \n", n);
    puts("-----------------------");
    for (i = 0; i < n; i++)
    {
        printf(" bus[%d] = No. %d \n", i, bus[i]);
    }
    puts("-----------------------");

    return 0;

    * There are [5] bus *
    -----------------------
    bus[0] = No. 111
    bus[1] = No. 22
    bus[2] = No. 333
    bus[3] = No. 444
    bus[4] = No. 555
    -----------------------
    */
}
```
