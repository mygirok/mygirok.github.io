```c
#include <stdio.h>

int main(void)
{
    puts("--------------------");
    puts(" conversion specification ");

    puts("--------------------");
    puts(" %d : decimal ");

    puts("--------------------");
    puts(" %c : character ");

    puts("--------------------");
    puts(" %s : string ");

    puts("--------------------");
    puts(" %f : float(4byte)");

    puts("--------------------");
    puts(" %lf : long float(8byte)");

    return 0;
}
```
